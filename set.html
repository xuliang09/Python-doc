set函数手册

集合遍历

例子
    >>> a = {1, 2}
    >>> for i in a:
    >>>     print(i)

|

语法
    set() | set()

功能
    求两个集合的并集

参数
    两个集合

例子
    >>> a = {1, 2, 3}
    >>> b = {1, 2, 4}
    >>> a | b
    {1, 2, 3, 4}

-

语法
    set() - set()

功能
    求两个集合的差集

参数
    两个集合

例子
    >>> a = {1, 2, 3}
    >>> b = {1, 2, 4}
    >>> a - b
    {3}
    >>> b - a
    {4}

&

语法
    set() & set()

功能
    求两个集合的交集

参数
    两个集合

例子
    >>> a = {1, 2, 3}
    >>> b = {1, 2, 4}
    >>> a & b
    {1, 2}

!=

语法
    set() != set()

功能
    判断两个集合的元素是否不一样

参数
    两个集合

例子
    >>> a = {1, 2}
    >>> b = {1, 2}
    >>> a != b
    False

==

语法
    set() == set()

功能
    判断两个集合的元素是否一样

参数
    两个集合

例子
    >>> a = {1, 2}
    >>> b = {1, 2}
    >>> a == b
    True

in

语法
    a in set()

功能
    元素a是否在集合中

参数
    一个元素，一个集合

例子
    >>> a = {1, 2}
    >>> 1 in a
    True

not in

语法
    a not in set()

功能
    元素a是否不在集合中

参数
    一个元素，一个集合

例子
    >>> a = {1, 2}
    >>> 1 not in a
    False
