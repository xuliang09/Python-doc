def cal_vec_dis(vector1, vector2):
     """
     calculate distance between vector
    
     :param vector1: fmt: list
     :param vector2: fmt: list
     :return: distance, fmt: double
     :raises AssertionError: length of vector1 and vector2 not equal
     """
     assert len(vector1) == len(vector2)
    
     dis = 0.0
     for i in range(len(vector1)):
     dis += math.pow(vector1[i] - vector2[i], 2)
     return math.sqrt(dis)
